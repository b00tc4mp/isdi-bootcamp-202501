# Declarar variables

## let
let (valor que puede cambiar) 
let mensaje = "Hola, mundo!";
console.log(mensaje); 
// Salida: Hola, mundo!

## const
const (valor que no cambia)
const saludo = "Hola, mundo!";
console.log(saludo); 
// Salida: Hola, mundo!


# Tipos de datos básicos
let numero = 5;
console.log(numero); 
// Salida: 5

let texto = "Hola, mundo!";
console.log(texto); 
// Salida: Hola, mundo!

let esVerdad = true;
console.log(esVerdad); 
// Salida: true

let esFalso = false;
console.log(esFalso); 
// Salida: false

let miVariable;
console.log(miVariable); 
// Salida: undefined

let variable = null;
console.log(variable); 
// Salida: null

# Operadores aritméticos
## aritméticos (+, -, *, /, %)
let a = 5;
let b = 3;
let suma = a + b;
console.log(suma); 
// Salida: 8

let a = 10;
let b = 3;
let resta = a - b;
console.log(resta); 
// Salida: 7

let a = 5;
let b = 3;
let multiplicacion = a * b;
console.log(multiplicacion); 
// Salida: 15

let a = 10;
let b = 2;
let division = a / b;
console.log(division); 
// Salida: 5

let a = 10;
let b = 3;
let residuo = a % b;
console.log(residuo); 
// Salida: 1 (10 dividido entre 3 da 3 con residuo 1)
